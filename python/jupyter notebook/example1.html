def getNaiveBayesProbabilities(obj, data, cl):
    p = [0.0, 0.0]
    # TODo
    s1=1
    s0=1
    for i in range(len(cl)):
        if cl[i]==1:
                s1+=1
        else:
                s0+=1
    p[0]=s0/(len(cl)+1)
    p[1]=s1/(len(cl)+1)
    for i in range(len(obj)):
        s1=0
        s0=0
        l1=0
        l0=0
        for j in range(len(data)):
            if cl[j]==1:
                s1=s1+data[j][i]
                l1+=1
            else:
                s0=s0+data[j][i]
                l0+=1
        s0=float(s0/l0)
        s1=float(s1/l1)
        if obj[i]==0:
            s1=1-s1
            s0=1-s0
        if s1==0:
            s1=0.1
        if s0==0:
            s0=0.1
        p[0]=p[0]*s0
        p[1]=p[1]*s1
        
    return p

def tabelka(tab,clas,bez):
    tab2=copy.deepcopy(tab)
    for i in range(len(clas)):
        for j in range(len(clas[i])):
            tab2[clas[i][j]].append(i)
    for j in range(len(bez)):
            del tab2[bez[j]-j]
    return(tab2)
    
def sortu(a,tab):
    t=0

    for i in range(len(tab2)):
